## 10.1 DAO 治理機制

### DAO 治理架構設計

DAO (Decentralized Autonomous Organization) 透過智能合約和代幣持有者投票實現去中心化治理。

#### 提案類型

```python
from enum import Enum
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Callable
import time

class ProposalType(Enum):
    PARAMETER_CHANGE = "parameter_change"     # 修改協議參數
    TREASURY_SPENDING = "treasury_spending"   # 資金撥付
    PROTOCOL_UPGRADE = "protocol_upgrade"     # 合約升級
    MEMBER_MANAGEMENT = "member_management"   # 成員管理
    EMERGENCY_ACTION = "emergency_action"     # 緊急操作

class ProposalStatus(Enum):
    DRAFT = "draft"           # 草稿（未提交）
    ACTIVE = "active"         # 投票進行中
    QUEUED = "queued"         # 通過，等待執行（時間鎖）
    EXECUTED = "executed"     # 已執行
    DEFEATED = "defeated"     # 未通過
    CANCELLED = "cancelled"   # 已取消
    EXPIRED = "expired"       # 執行時間已過期

class VoteType(Enum):
    FOR = "for"
    AGAINST = "against"
    ABSTAIN = "abstain"
```

#### 治理參數

```python
@dataclass
class GovernanceParameters:
    proposal_threshold: float = 0.01    # 提案門檻：1% 投票權
    quorum: float = 0.04               # 法定人數：4% 總供應量需參與
    approval_threshold: float = 0.50   # 通過門檻：50% 贊成
    voting_delay: int = 86400          # 投票延遲：24 小時
    voting_period: int = 5 * 86400     # 投票期：5 天
    timelock_delay: int = 2 * 86400    # 時間鎖：48 小時
    execution_deadline: int = 7 * 86400  # 執行期限：7 天

# 各類提案的特殊參數（可覆蓋默認值）
PROPOSAL_TYPE_PARAMS = {
    ProposalType.EMERGENCY_ACTION: {
        'quorum': 0.10,          # 緊急操作需要更高法定人數
        'approval_threshold': 0.67,  # 需要 2/3 同意
        'voting_period': 86400,   # 只有 1 天投票期
        'timelock_delay': 0       # 無時間鎖（緊急）
    },
    ProposalType.PROTOCOL_UPGRADE: {
        'quorum': 0.08,
        'approval_threshold': 0.60,
        'timelock_delay': 7 * 86400  # 協議升級有 7 天時間鎖
    }
}
```

#### Proposal 數據結構

```python
@dataclass
class Vote:
    voter: str
    vote_type: VoteType
    voting_power: int       # 快照時的投票權（代幣數量）
    timestamp: int
    reason: str = ''        # 可選的投票理由

@dataclass
class Proposal:
    proposal_id: str
    proposal_type: ProposalType
    title: str
    description: str
    proposer: str
    
    # 可執行的操作（targets, values, calldatas）
    target_contracts: List[str]    # 調用的合約地址
    call_values: List[int]         # 附帶的 ETH 金額
    calldatas: List[bytes]         # 調用數據（函數選擇器 + 參數）
    
    # 時間管理
    created_at: int = field(default_factory=lambda: int(time.time()))
    voting_start: int = 0          # 投票開始時間（created_at + voting_delay）
    voting_end: int = 0            # 投票結束時間
    queued_at: int = 0             # 進入執行佇列的時間
    executed_at: int = 0           # 執行時間
    
    # 狀態
    status: ProposalStatus = ProposalStatus.DRAFT
    
    # 投票統計
    votes_for: int = 0
    votes_against: int = 0
    votes_abstain: int = 0
    vote_records: List[Vote] = field(default_factory=list)
    
    # 快照（防止投票期間買入代幣影響結果）
    snapshot_block: int = 0        # 計算投票權的區塊高度
    total_supply_at_snapshot: int = 0
```

#### DAOGovernance 主類

```python
class DAOGovernance:
    def __init__(self, token_contract, timelock_contract, params: GovernanceParameters = None):
        self.token = token_contract           # 治理代幣合約
        self.timelock = timelock_contract     # 時間鎖合約
        self.params = params or GovernanceParameters()
        self.proposals: Dict[str, Proposal] = {}
        self.proposal_counter = 0
        self.delegate_registry: Dict[str, str] = {}  # voter → delegatee
    
    def create_proposal(self, proposer: str, proposal_type: ProposalType,
                         title: str, description: str,
                         targets: List[str], values: List[int], calldatas: List[bytes]) -> str:
        """創建提案（需要持有足夠的投票權）"""
        
        # 驗證提案者的投票權
        proposer_power = self.get_voting_power(proposer)
        total_supply = self.token.total_supply()
        
        required_power = int(total_supply * self.params.proposal_threshold)
        if proposer_power < required_power:
            raise InsufficientVotingPower(
                f"需要 {required_power} 投票權，當前持有 {proposer_power}"
            )
        
        # 驗證操作列表長度一致
        assert len(targets) == len(values) == len(calldatas), "操作列表長度不一致"
        
        # 創建提案
        self.proposal_counter += 1
        proposal_id = f"PROP-{self.proposal_counter:04d}"
        
        now = int(time.time())
        voting_start = now + self.params.voting_delay
        voting_end = voting_start + self.params.voting_period
        
        proposal = Proposal(
            proposal_id=proposal_id,
            proposal_type=proposal_type,
            title=title,
            description=description,
            proposer=proposer,
            target_contracts=targets,
            call_values=values,
            calldatas=calldatas,
            created_at=now,
            voting_start=voting_start,
            voting_end=voting_end,
            snapshot_block=self._get_current_block(),
            total_supply_at_snapshot=total_supply
        )
        
        self.proposals[proposal_id] = proposal
        proposal.status = ProposalStatus.ACTIVE
        
        self._emit_event('ProposalCreated', {
            'id': proposal_id, 'proposer': proposer, 'title': title,
            'voting_start': voting_start, 'voting_end': voting_end
        })
        
        return proposal_id
    
    def cast_vote(self, voter: str, proposal_id: str, 
                  vote_type: VoteType, reason: str = '') -> int:
        """
        投票
        投票權基於快照區塊時的代幣餘額（防止投票期間購入代幣影響結果）
        """
        proposal = self.proposals.get(proposal_id)
        if proposal is None:
            raise ValueError(f"提案 {proposal_id} 不存在")
        
        now = int(time.time())
        if now < proposal.voting_start:
            raise ValueError("投票尚未開始")
        if now > proposal.voting_end:
            raise ValueError("投票已結束")
        if proposal.status != ProposalStatus.ACTIVE:
            raise ValueError(f"提案狀態不允許投票：{proposal.status}")
        
        # 檢查是否重複投票
        if any(v.voter == voter for v in proposal.vote_records):
            raise ValueError("已經投過票了")
        
        # 獲取快照時的投票權（含委託）
        voting_power = self.get_voting_power_at_snapshot(voter, proposal.snapshot_block)
        
        if voting_power == 0:
            raise ValueError("無投票權（快照時未持有代幣）")
        
        # 記錄投票
        vote = Vote(
            voter=voter,
            vote_type=vote_type,
            voting_power=voting_power,
            timestamp=now,
            reason=reason
        )
        proposal.vote_records.append(vote)
        
        # 累計投票結果
        if vote_type == VoteType.FOR:
            proposal.votes_for += voting_power
        elif vote_type == VoteType.AGAINST:
            proposal.votes_against += voting_power
        else:
            proposal.votes_abstain += voting_power
        
        self._emit_event('VoteCast', {
            'voter': voter, 'proposal_id': proposal_id,
            'vote_type': vote_type.value, 'voting_power': voting_power,
            'reason': reason
        })
        
        return voting_power
    
    def finalize_proposal(self, proposal_id: str) -> ProposalStatus:
        """投票期結束後，計算結果並更新狀態"""
        proposal = self.proposals.get(proposal_id)
        if proposal is None:
            raise ValueError(f"提案不存在：{proposal_id}")
        
        if int(time.time()) < proposal.voting_end:
            raise ValueError("投票尚未結束")
        
        total_votes = proposal.votes_for + proposal.votes_against + proposal.votes_abstain
        
        # 獲取該提案類型的特殊參數（若有）
        type_params = PROPOSAL_TYPE_PARAMS.get(proposal.proposal_type, {})
        quorum = type_params.get('quorum', self.params.quorum)
        approval = type_params.get('approval_threshold', self.params.approval_threshold)
        
        # 計算法定人數（基於快照時的總供應量）
        quorum_required = int(proposal.total_supply_at_snapshot * quorum)
        
        if total_votes < quorum_required:
            proposal.status = ProposalStatus.DEFEATED
            self._emit_event('ProposalDefeated', {'id': proposal_id, 'reason': 'quorum_not_met'})
            return ProposalStatus.DEFEATED
        
        # 計算通過率（不含棄權票）
        decisive_votes = proposal.votes_for + proposal.votes_against
        if decisive_votes == 0:
            proposal.status = ProposalStatus.DEFEATED
            return ProposalStatus.DEFEATED
        
        approval_rate = proposal.votes_for / decisive_votes
        
        if approval_rate >= approval:
            proposal.status = ProposalStatus.QUEUED
            proposal.queued_at = int(time.time())
            self._emit_event('ProposalQueued', {'id': proposal_id, 'approval_rate': approval_rate})
            return ProposalStatus.QUEUED
        else:
            proposal.status = ProposalStatus.DEFEATED
            self._emit_event('ProposalDefeated', {
                'id': proposal_id, 
                'reason': 'approval_not_met',
                'approval_rate': approval_rate
            })
            return ProposalStatus.DEFEATED
    
    def execute_proposal(self, executor: str, proposal_id: str) -> bool:
        """
        執行已通過的提案（時間鎖過後）
        任何人都可以調用執行，但必須在執行期限內
        """
        proposal = self.proposals.get(proposal_id)
        if proposal.status != ProposalStatus.QUEUED:
            raise ValueError(f"提案狀態不允許執行：{proposal.status}")
        
        now = int(time.time())
        
        # 獲取時間鎖延遲
        type_params = PROPOSAL_TYPE_PARAMS.get(proposal.proposal_type, {})
        timelock = type_params.get('timelock_delay', self.params.timelock_delay)
        
        earliest_execution = proposal.queued_at + timelock
        if now < earliest_execution:
            remaining = earliest_execution - now
            raise ValueError(f"時間鎖未到期，剩餘 {remaining:.0f} 秒")
        
        execution_deadline = earliest_execution + self.params.execution_deadline
        if now > execution_deadline:
            proposal.status = ProposalStatus.EXPIRED
            raise ValueError("超過執行期限，提案已過期")
        
        # 執行所有操作
        results = []
        for target, value, calldata in zip(
            proposal.target_contracts, proposal.call_values, proposal.calldatas
        ):
            try:
                result = self._call_contract(target, value, calldata)
                results.append(result)
            except Exception as e:
                raise ExecutionError(f"操作執行失敗：{e}")
        
        proposal.status = ProposalStatus.EXECUTED
        proposal.executed_at = now
        
        self._emit_event('ProposalExecuted', {
            'id': proposal_id, 'executor': executor, 'timestamp': now
        })
        
        return True
```

#### 投票權委託系統

```python
class VotingPowerDelegation:
    """投票權委託（如 Compound 的委託模型）"""
    
    def __init__(self, token_contract):
        self.token = token_contract
        self.delegates: Dict[str, str] = {}        # voter → delegatee
        self.checkpoints: Dict[str, List[tuple]] = {}  # address → [(block, power)]
    
    def delegate(self, delegator: str, delegatee: str):
        """委託投票權"""
        old_delegatee = self.delegates.get(delegator)
        
        if old_delegatee == delegatee:
            return  # 無變化
        
        delegator_balance = self.token.balance_of(delegator)
        
        # 更新舊委託人的投票權
        if old_delegatee:
            self._write_checkpoint(
                old_delegatee,
                self._get_current_votes(old_delegatee) - delegator_balance
            )
        
        # 更新新委託人的投票權
        self._write_checkpoint(
            delegatee,
            self._get_current_votes(delegatee) + delegator_balance
        )
        
        self.delegates[delegator] = delegatee
        
        self._emit_event('DelegateChanged', {
            'delegator': delegator,
            'from_delegate': old_delegatee,
            'to_delegate': delegatee
        })
    
    def get_prior_votes(self, account: str, block_number: int) -> int:
        """獲取特定區塊時的投票權（快照查詢）"""
        checkpoints = self.checkpoints.get(account, [])
        
        if not checkpoints or block_number < checkpoints[0][0]:
            return 0
        
        if block_number >= checkpoints[-1][0]:
            return checkpoints[-1][1]
        
        # 二分搜索（checkpoints 按區塊號排序）
        lo, hi = 0, len(checkpoints) - 1
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if checkpoints[mid][0] <= block_number:
                lo = mid
            else:
                hi = mid - 1
        
        return checkpoints[lo][1]
    
    def _write_checkpoint(self, account: str, new_votes: int):
        """記錄投票權快照"""
        current_block = self._get_current_block()
        
        if account not in self.checkpoints:
            self.checkpoints[account] = []
        
        checkpoints = self.checkpoints[account]
        
        if checkpoints and checkpoints[-1][0] == current_block:
            checkpoints[-1] = (current_block, new_votes)
        else:
            checkpoints.append((current_block, new_votes))
```

#### Timelock 合約

```python
class TimelockController:
    """
    時間鎖合約：確保社區有足夠時間審閱和反對治理決策
    來源自 OpenZeppelin TimelockController
    """
    
    PENDING = b'\x01'
    DONE = b'\x02'
    
    def __init__(self, min_delay: int, proposers: List[str], executors: List[str]):
        self.min_delay = min_delay
        self.proposers = set(proposers)
        self.executors = set(executors)
        self.timestamps: Dict[bytes, int] = {}  # operation_id → scheduled_time
    
    def schedule(self, target: str, value: int, calldata: bytes, 
                  predecessor: bytes, salt: bytes, delay: int) -> bytes:
        """安排操作（在 delay 秒後可執行）"""
        if delay < self.min_delay:
            raise ValueError(f"延遲過短：{delay} < {self.min_delay}")
        
        operation_id = self._hash_operation(target, value, calldata, predecessor, salt)
        
        if self.timestamps.get(operation_id) is not None:
            raise ValueError("操作已被安排")
        
        self.timestamps[operation_id] = int(time.time()) + delay
        
        self._emit_event('CallScheduled', {
            'id': operation_id.hex(), 'target': target, 'delay': delay
        })
        
        return operation_id
    
    def execute(self, target: str, value: int, calldata: bytes,
                predecessor: bytes, salt: bytes) -> bytes:
        """執行已安排的操作"""
        operation_id = self._hash_operation(target, value, calldata, predecessor, salt)
        
        scheduled_time = self.timestamps.get(operation_id)
        if scheduled_time is None:
            raise ValueError("操作未被安排")
        
        if int(time.time()) < scheduled_time:
            remaining = scheduled_time - int(time.time())
            raise ValueError(f"時間鎖未到期，剩餘 {remaining:.0f} 秒")
        
        # 驗證前置條件（predecessor 必須已執行）
        if predecessor and self.timestamps.get(predecessor) != int.from_bytes(self.DONE, 'big'):
            raise ValueError("前置操作未完成")
        
        # 執行
        result = self._call(target, value, calldata)
        
        # 標記為已完成
        self.timestamps[operation_id] = int.from_bytes(self.DONE, 'big')
        
        self._emit_event('CallExecuted', {'id': operation_id.hex(), 'target': target})
        return result
    
    def cancel(self, operation_id: bytes, canceller: str):
        """取消待執行的操作（需要 Guardian 權限）"""
        if canceller not in self.proposers:
            raise PermissionError("無取消權限")
        
        if self.timestamps.get(operation_id) is None:
            raise ValueError("操作不存在")
        
        del self.timestamps[operation_id]
        self._emit_event('Cancelled', {'id': operation_id.hex()})
    
    def _hash_operation(self, target, value, calldata, predecessor, salt) -> bytes:
        data = (
            target.encode() + 
            value.to_bytes(32, 'big') + 
            calldata + 
            (predecessor or b'\x00' * 32) + 
            salt
        )
        return hashlib.sha256(data).digest()
```

```python
class GovernanceGuardian:
    """緊急安全防護：多簽守護者可以緊急取消惡意提案"""
    
    def __init__(self, multisig_address: str, timelock: TimelockController):
        self.multisig = multisig_address
        self.timelock = timelock
    
    def emergency_cancel(self, operation_id: bytes, reason: str):
        """緊急取消已通過但尚未執行的惡意提案"""
        # 需要多簽守護者授權
        if not self._verify_multisig_authorization():
            raise PermissionError("需要多簽守護者授權")
        
        self.timelock.cancel(operation_id, self.multisig)
        
        self._emit_emergency_event({
            'action': 'emergency_cancel',
            'operation_id': operation_id.hex(),
            'reason': reason,
            'timestamp': int(time.time())
        })
```

#### DAO 治理最佳實踐

| 維度 | 建議 | 原因 |
|------|------|------|
| 投票門檻 | 1-5% 提案 / 4-10% 法定人數 | 防止無效提案，但保持參與可及性 |
| 時間設計 | 1 天延遲 + 5 天投票 + 2 天時間鎖 | 充分討論時間，防止閃電攻擊 |
| 投票類型 | FOR / AGAINST / ABSTAIN | ABSTAIN 幫助達到法定人數但不表態 |
| 委託機制 | 支持投票委託 | 提高活躍參與率，允許專業治理代理 |
| 緊急機制 | 多簽守護者 + 暫停機制 | 應對緊急安全事件 |
| 透明度 | 完整的事件日誌 + 鏈上數據 | 可審計性，建立社區信任 |

> **🔄 技術更新（2025+）**：
> - **Snapshot（鏈外投票）**：已成為 DAO 溫度測試（Temperature Check）的標準工具，利用離鏈簽名+中心化存儲（IPFS）實現免 Gas 投票，大幅提升參與率。治理流程通常：Snapshot 投票 → 鏈上執行，而非全部鏈上。
> - **MakerDAO → Sky 改名（2024）**：MakerDAO 推行「Endgame」計劃，更名為 Sky，DAI 升級為 USDS，MKR 升級為 SKY，並推出子 DAO 架構，是大型 DAO 治理演化的重要案例。
> - **樂觀治理（Optimistic Governance）**：部分 DAO 採用「默認通過+否決機制」，提案若無足夠反對票則自動執行，降低低活躍率導致的治理停滯。
> - **Tally / Boardroom**：成熟的 DAO 治理界面工具，整合多個協議的提案、投票、委託功能，成為 DAO 參與者的標準入口。
> - **AI + DAO**：部分 DAO 開始嘗試 AI Agent 作為治理參與者，分析提案影響、代替持有者自動投票，是 2025 年的新興實驗。
